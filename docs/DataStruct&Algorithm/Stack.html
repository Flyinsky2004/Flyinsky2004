栈 ​

介绍

当谈到数据结构时，栈（Stack）是一种常见而重要的结构之一。栈是一种具有特定限制的线性数据结构，它遵循“先进后出”（Last In, First Out，简称LIFO）的原则。这意味着最后放入栈的元素将是第一个被取出的，而最先放入的元素则会被推迟到后面。入栈可以理解成往一口井中放入一袋沙子，出栈则是将离地面最近的那袋沙子(井中最上面那袋沙子，也就是最晚入井的那袋沙子)从井里提出来。

1.栈的定义 ​

栈是一种具有以下两个主要操作的抽象数据类型：

  1. Push（入栈）： 将元素添加到栈的顶部。
  2. Pop（出栈）： 从栈的顶部移除元素。

栈还可以包括一个用于查看栈顶元素但不移除它的操作，通常称为Peek或Top。这些操作使栈成为解决许多问题的有力工具。

2.基本数据结构定义及增删改查 ​

栈的实现可以基于数组或链表。在这里，我们将关注数组实现的栈。

c
#include <stdio.h>
#include <stdlib.h>

// 定义栈节点
typedef struct StackNode {
    int data;
    struct StackNode* next;
} StackNode;

// 定义栈结构
typedef struct {
    StackNode* top;
} Stack;

// 初始化栈
void initStack(Stack* stack) {
    stack->top = NULL;
}

// 将元素推入栈
void push(Stack* stack, int element) {
    StackNode* newNode = (StackNode*)malloc(sizeof(StackNode));
    if (newNode == NULL) {
        fprintf(stderr, "内存分配失败\n");
        exit(EXIT_FAILURE);
    }
    newNode->data = element;
    newNode->next = stack->top;
    stack->top = newNode;
}

// 从栈中弹出元素
void pop(Stack* stack) {
    if (!isEmpty(stack)) {
        StackNode* temp = stack->top;
        stack->top = stack->top->next;
        free(temp);
    }
}

// 查看栈顶元素
int top(const Stack* stack) {
    if (!isEmpty(stack)) {
        return stack->top->data;
    } else {
        // 返回一个默认值或抛出异常，取决于实际需求
        // 这里简单返回一个默认值
        return 0;
    }
}

// 检查栈是否为空
int isEmpty(const Stack* stack) {
    return stack->top == NULL;
}

// 获取栈的大小
size_t size(const Stack* stack) {
    size_t count = 0;
    StackNode* current = stack->top;
    while (current != NULL) {
        count++;
        current = current->next;
    }
    return count;
}

// 释放栈的内存
void destroyStack(Stack* stack) {
    while (!isEmpty(stack)) {
        pop(stack);
    }
}

int main() {
    Stack myStack;
    initStack(&myStack);

    push(&myStack, 42);
    push(&myStack, 23);
    push(&myStack, 7);

    printf("栈顶元素: %d\n", top(&myStack));
    printf("栈的大小: %zu\n", size(&myStack));

    pop(&myStack);

    printf("弹出一个元素后，栈顶元素: %d\n", top(&myStack));
    printf("栈的大小: %zu\n", size(&myStack));

    destroyStack(&myStack);

    return 0;
}

3.栈的应用 ​

(1).括号匹配 ​

介绍

括号匹配是一种常见的算法问题，通常在编程中会涉及到检查表达式或代码中的括号是否正确匹配。这个问题的解决方法之一是使用栈来追踪左括号，并在遇到右括号时检查是否与栈顶的左括号匹配。

I. 基本思想 ​

  • 创建一个栈来存储遇到的左括号。
  • 遍历表达式中的每个字符。
    • 如果是左括号（'(', '[', '{'等），将其推入栈中。
    • 如果是右括号（')', ']', '}'等），检查栈顶是否有与之匹配的左括号。
      • 如果有匹配的左括号，将栈顶的左括号弹出，继续检查下一个字符。
      • 如果没有匹配的左括号，说明括号不匹配。

II.代码实现 ​

这是一个C语言实现的将栈应用于括号匹配算法

c
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100

typedef struct {
    char data[MAX_SIZE];
    int top;
} Stack;

void initStack(Stack* stack) {
    stack->top = -1;
}

int isEmpty(const Stack* stack) {
    return stack->top == -1;
}

void push(Stack* stack, char element) {
    if (stack->top < MAX_SIZE - 1) {
        stack->data[++stack->top] = element;
    } else {
        fprintf(stderr, "栈溢出\n");
        exit(EXIT_FAILURE);
    }
}

char pop(Stack* stack) {
    if (!isEmpty(stack)) {
        return stack->data[stack->top--];
    } else {
        fprintf(stderr, "栈为空\n");
        exit(EXIT_FAILURE);
    }
}

int isMatchingPair(char left, char right) {
    return (left == '(' && right == ')') ||
           (left == '[' && right == ']') ||
           (left == '{' && right == '}');
}

int areParenthesesMatched(const char* expression) {
    Stack charStack;
    initStack(&charStack);

    int i = 0;
    char currentChar;

    while (expression[i] != '\0') {
        currentChar = expression[i];

        if (currentChar == '(' || currentChar == '[' || currentChar == '{') {
            push(&charStack, currentChar);
        } else if (currentChar == ')' || currentChar == ']' || currentChar == '}') {
            if (isEmpty(&charStack) || !isMatchingPair(pop(&charStack), currentChar)) {
                return 0; // 不匹配
            }
        }

        i++;
    }

    return isEmpty(&charStack); // 如果栈为空，说明所有左括号都有匹配的右括号
}

int main() {
    const char* expression1 = "{[()]}";
    const char* expression2 = "{[(])}";
    const char* expression3 = "((()))";

    if (areParenthesesMatched(expression1)) {
        printf("%s: 括号匹配\n", expression1);
    } else {
        printf("%s: 括号不匹配\n", expression1);
    }

    if (areParenthesesMatched(expression2)) {
        printf("%s: 括号匹配\n", expression2);
    } else {
        printf("%s: 括号不匹配\n", expression2);
    }

    if (areParenthesesMatched(expression3)) {
        printf("%s: 括号匹配\n", expression3);
    } else {
        printf("%s: 括号不匹配\n", expression3);
    }

    return 0;
}
